# Requirements for the Apis A2A inbound gateway (a2a_gateway.py).
#
# The SDK-agnostic core (a2a_executor.py, routing.py, the card builder/signer
# and authorize_caller in a2a_gateway.py) needs only the stdlib + cryptography
# (already used by lib/daemon_runtime/aauth_signer.py). The packages below are
# required only to actually *serve* the gateway over the wire.

# Official Linux Foundation A2A server/client SDK (Apache-2.0).
a2a-sdk>=0.3.0

# ASGI server used by a2a-sdk's Starlette application.
uvicorn>=0.30.0

# EC P-256 signing for AAuth + signed Agent Cards (also a lib/daemon_runtime dep).
cryptography>=42.0.0

# PyJWT for AAuth token minting (used by lib/daemon_runtime/aauth_signer.py).
PyJWT>=2.8.0
